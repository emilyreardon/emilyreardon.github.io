/* ─────────────────────────────

   IMAGE LOADER

───────────────────────────── */



function buildImage(basePath) {

  const img = document.createElement("img");

  img.src = basePath + ".jpg";

  img.onerror = () => {

    img.src = basePath + ".png";

  };

  return img;

}



/* ─────────────────────────────

   GIF GRID

───────────────────────────── */



function buildGifGrid(paths) {

  const grid = document.createElement("div");

  grid.className = "gif-grid";



  paths.forEach(p => {

    const img = document.createElement("img");

    img.src = p;

    grid.appendChild(img);

  });



  return grid;

}



/* ─────────────────────────────

   VIDEO RENDERER (MP4 FIX)

───────────────────────────── */



function buildVideo(src) {

  const video = document.createElement("video");

  video.src = src;

  video.controls = true;

  video.autoplay = false;

  video.loop = true;

  video.playsInline = true;

  return video;

}



/* ─────────────────────────────

   RENDER

───────────────────────────── */



function renderProjects() {

  const container = document.getElementById("projects-list");



  projects.forEach(p => {

    const card = document.createElement("article");

    card.className = "project-card";



    /* title */

    const title = document.createElement("div");

    title.className = "project-title";

    title.textContent = p.title;

    card.appendChild(title);



    /* media */

    if (p.videoPaths) {

      p.videoPaths.forEach(v => {

        card.appendChild(buildVideo(v));

      });

    }



    if (p.gifPaths) {

      card.appendChild(buildGifGrid(p.gifPaths));

    }



    if (p.imagePaths) {

      p.imagePaths.forEach(i => {

        card.appendChild(buildImage(i));

      });

    }



    /* byline */

    const byline = document.createElement("div");

    byline.className = "project-byline";

    byline.textContent = `${p.year} · ${p.tools}`;

    card.appendChild(byline);



    /* rationale */

    const r = document.createElement("div");

    r.className = "project-label";

    r.textContent = "Rationale";

    card.appendChild(r);



    const rText = document.createElement("div");

    rText.className = "project-body";

    rText.innerHTML = p.rationale;

    card.appendChild(rText);



    /* description */

    const d = document.createElement("div");

    d.className = "project-label";

    d.textContent = "Project Description";

    card.appendChild(d);



    const dText = document.createElement("div");

    dText.className = "project-body";

    dText.innerHTML = p.description;

    card.appendChild(dText);



    container.appendChild(card);

  });

}
